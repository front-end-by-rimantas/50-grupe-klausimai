function maxMin(arr1,arr2){
  //coding here...
  const arrDiff = [];
  
  for(let ii = 0; ii < arr2.length; ii++){
    arrDiff.push(Math.abs(arr2[ii] - arr1[ii]));   
  }
  const max = Math.max(...arrDiff);
  const mini = Math.min(...arrDiff);
  
  return [max, mini];
}
